HR,30000"""
        )
    print("departments.txt created")


def calculate_department_salary():
    salaries = {}

    with open("/tmp/departments.txt", "r") as file:
        for line in file:
            dept, salary = line.strip().split(",")
            salary = int(salary)
            if dept in salaries:
                salaries[dept] += salary
            else:
                salaries[dept] = salary

    with open("/tmp/dept_salary.txt", "w") as file:
        for dept, total in salaries.items():
            file.write(f"{dept}={total}\n")

    print("Department salaries calculated")


def generate_department_report():
    with open("/tmp/dept_salary.txt", "r") as file:
        content = file.read()

    with open("/tmp/department_report.txt", "w") as file:
        file.write("Department Salary Report\n")
        file.write(content)

    print("Department report generated")

        )

def calculate_order_value():
    revenues = {}

    with open("/tmp/orders.csv", "r") as file:
        reader = csv.DictReader(file)
        for row in reader:
            revenue = int(row["quantity"]) * int(row["price"])
            revenues[row["product"]] = revenue

    with open("/tmp/order_values.txt", "w") as file:
        for product, revenue in revenues.items():
            file.write(f"{product},{revenue}\n")

    print("Order values calculated")


def generate_sales_report():
    revenues = {}

    with open("/tmp/order_values.txt", "r") as file:
        for line in file:
            product, revenue = line.strip().split(",")
            revenues[product] = int(revenue)

    total = sum(revenues.values())
    highest = max(revenues, key=revenues.get)

    with open("/tmp/sales_report.txt", "w") as file:
        file.write("Sales Report\n")
        for product, revenue in revenues.items():
            file.write(f"{product} = {revenue}\n")
        file.write(f"Total Revenue = {total}\n")
        file.write(f"Highest Selling Product = {highest}")

    print("Sales report generated")

def calculate_balance():
    deposits = 0
    withdrawals = 0

    with open("/tmp/transactions.txt", "r") as file:
        for line in file:
            txn_type, amount = line.strip().split(",")
            amount = int(amount)
            if txn_type == "Deposit":
                deposits += amount
            else:

    with open("/tmp/balance.txt", "w") as file:
        file.write(f"{deposits},{withdrawals},{balance}")
